func NewUUID() (string, error) { // uuid v4 vía crypto/rand
  var b [16]byte
  if _, err := rand.Read(b[:]); err != nil { return "", err }
  b[6] = (b[6] & 0x0f) | 0x40; b[8] = (b[8] & 0x3f) | 0x80
  return fmt.Sprintf("%x-%x-%x-%x-%x", b[0:4], b[4:6], b[6:8], b[8:10], b[10:16]), nil
}

func CCHome(home, profile string) (string, error) {
  if profile == "default" {
    uh, err := os.UserHomeDir(); if err != nil { return "", err }
    return filepath.Join(uh, ".claude"), nil
  }
  return filepath.Join(home, "profiles", profile, "cc-home"), nil
}

04CopyTranscript (forward, mismo uuid)

Mod: core/transcript.go

Copia cruda conservando el nombre (uuid). sessionId ya == uuid, cwd mismo repo → sin reescritura. Colisión: idéntico → overwrite; distinto → error salvo force.

func CopyTranscript(srcPath, dstDir string, force bool) (string, error) {
  data, _ := os.ReadFile(srcPath); os.MkdirAll(dstDir, 0o755)
  dstPath := filepath.Join(dstDir, filepath.Base(srcPath))
  if existing, err := os.ReadFile(dstPath); err == nil {
    if !bytes.Equal(existing, data) && !force { return "", fmt.Errorf("colisión...") }
  }
  return dstPath, os.WriteFile(dstPath, data, 0o644)
}

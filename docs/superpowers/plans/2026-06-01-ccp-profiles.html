§ 00Arquitectura & estructura de archivos

Goal: rebrand dsctl→ccp y convertir el toggle deepseek↔official en un motor de perfiles con nombre que enruta cada carpeta a una cuenta oficial o a un provider DeepSeek, todo configurable por CLI.

Arquitectura: libs de funciones puras (lib/paths.sh resolver, lib/profiles.sh storage, lib/env.sh emisor de delta) sourced por un binario bin/ccp. El binario corre en hijo y no muta el shell; la función ccp() y el hook precmd hacen eval del delta emitido — único punto de mutación de env.

Convenciones fijadas (usadas en todas las tareas)

  • Raíz config: CCP_HOME (default ~/.config/ccp).
  • Perfil: $CCP_HOME/profiles/<name>/{meta, api_key (600), cc-home/}.
  • rules.tsv = path<TAB>profile; profiles.tsv = name<TAB>type.
  • default = perfil reservado (nunca en disco) = fallback.
  • meta = key=value parseado (NUNCA sourced).
  • Vars manejadas (set exacto en lib/env.sh): CLAUDE_CONFIG_DIR ANTHROPIC_BASE_URL ANTHROPIC_AUTH_TOKEN ANTHROPIC_MODEL ANTHROPIC_DEFAULT_{OPUS,SONNET,HAIKU}_MODEL CLAUDE_CODE_SUBAGENT_MODEL CLAUDE_CODE_EFFORT_LEVEL CCP_PROFILE.

§ 02Helpers de path (rename ds_*→ccp_*)

Modificar: lib/paths.sh · tests/run.sh

Rename mecánico (cuerpos idénticos) de ds_norm_path/ds_is_ancestor/ds_depth → ccp_*, fijando los nombres nuevos. Tests: tilde expande, .. colapsa, raíz, ancestro, profundidad.

ccp_norm_path(){ # ~ inicial, a absoluto, colapsa . y .. textual
  local p="$1"; [[ -z "$p" ]] && return 1
  [[ "$p" == "~"* ]] && p="${p/#\~/$HOME}"; [[ "$p" != /* ]] && p="$(pwd)/$p"
  local out=() part; local IFS='/'
  for part in $p; do case "$part" in ''|.) continue;; ..) [[ ${#out[@]} -gt 0 ]] && unset 'out[${#out[@]}-1]';; *) out+=("$part");; esac; done
  local j=""; for part in "${out[@]}"; do j="$j/$part"; done; printf '%s' "${j:-/}"; }
ccp_is_ancestor(){ local base="$1" path="$2"; [[ "$base" == "/" ]] && return 0
  [[ "$path" == "$base" ]] && return 0; [[ "$path" == "$base"/* ]] && return 0; return 1; }
ccp_depth(){ local p="$1" d=0 part; local IFS='/'; for part in $p; do [[ -n "$part" ]] && d=$((d+1)); done; printf '%d' "$d"; }

§ 03Resolver → nombre de perfil

Modificar: lib/paths.sh · tests/run.sh

Formato de regla pasa a path<TAB>profile. ccp_resolve devuelve el nombre del perfil más específico (o default). ccp_rule_set reemplaza por path; ccp_rule_del borra. Tests: vacío→default, most-specific gana, carve-out default, replace no-append, del.

ccp_resolve(){ # <path> <rules_file> → nombre (default si nada aplica)
  local query rules_file; query="$(ccp_norm_path "$1")" || { printf 'default'; return; }
  rules_file="$2"; [[ -f "$rules_file" ]] || { printf 'default'; return; }
  local best_depth=-1 best_profile="default" path profile depth
  while IFS=$'\t' read -r path profile; do
    [[ -z "$path" || "$path" == \#* || -z "$profile" ]] && continue
    if ccp_is_ancestor "$path" "$query"; then depth="$(ccp_depth "$path")"
      if (( depth > best_depth )); then best_depth="$depth"; best_profile="$profile"; fi; fi
  done < "$rules_file"; printf '%s' "$best_profile"; }
ccp_rule_set(){ local path profile="$2" rules_file="$3"; path="$(ccp_norm_path "$1")" || return 1
  mkdir -p "$(dirname "$rules_file")"; touch "$rules_file"; local tmp; tmp="$(mktemp)"
  awk -F'\t' -v p="$path" '$1 != p' "$rules_file" > "$tmp" && mv "$tmp" "$rules_file"
  printf '%s\t%s\n' "$path" "$profile" >> "$rules_file"; }
ccp_rule_del(){ local path rules_file="$2"; path="$(ccp_norm_path "$1")" || return 1
  [[ -f "$rules_file" ]] || return 0; local tmp; tmp="$(mktemp)"
  awk -F'\t' -v p="$path" '$1 != p' "$rules_file" > "$tmp" && mv "$tmp" "$rules_file"; }

_ccp_profiles_dir(){ printf '%s/profiles' "$1"; }

ccp_profile_exists(){ [[ -f "$(_ccp_profile_dir "$1" "$2")/meta" ]]; }

ccp_profile_get(){ # ccp_home name key — imprime tras el primer '='
  local meta; meta="$(_ccp_profile_dir "$1" "$2")/meta"; [[ -f "$meta" ]] || return 1
  awk -F= -v k="$3" '$1==k{sub(/^[^=]*=/,"");print;found=1;exit} END{exit !found}' "$meta"; }
_ccp_index_set(){ local idx="$1/profiles.tsv"; mkdir -p "$1"; touch "$idx"; local tmp; tmp="$(mktemp)"
  awk -F'\t' -v n="$2" '$1 != n' "$idx" > "$tmp" && mv "$tmp" "$idx"; printf '%s\t%s\n' "$2" "$3" >> "$idx"; }
ccp_profile_add_official(){ local d; d="$(_ccp_profile_dir "$1" "$2")"; mkdir -p "$d/cc-home"
  printf 'type=official\n' > "$d/meta"; _ccp_index_set "$1" "$2" official; }
ccp_profile_add_deepseek(){ # home name base_url pro flash effort
  local d; d="$(_ccp_profile_dir "$1" "$2")"; mkdir -p "$d"
  { printf 'type=deepseek\n'; printf 'base_url=%s\n' "$3"; printf 'model_pro=%s\n' "$4"
    printf 'model_flash=%s\n' "$5"; printf 'effort=%s\n' "$6"; } > "$d/meta"

Fase 3 · Env delta

§ 06env.sh — ccp_env_delta

Crear: lib/env.sh · Modificar: tests/run.sh

Emite delta eval-able: (1) unset de todas las vars manejadas, (2) export del perfil. Valores con %q. Tests evalúan en subshell y verifican efecto.

CCP_MANAGED_VARS="CLAUDE_CONFIG_DIR ANTHROPIC_BASE_URL ANTHROPIC_AUTH_TOKEN ANTHROPIC_MODEL ANTHROPIC_DEFAULT_OPUS_MODEL ANTHROPIC_DEFAULT_SONNET_MODEL ANTHROPIC_DEFAULT_HAIKU_MODEL CLAUDE_CODE_SUBAGENT_MODEL CLAUDE_CODE_EFFORT_LEVEL CCP_PROFILE"
ccp_env_delta(){ local home="$1" profile="$2"
  printf 'unset %s\n' "$CCP_MANAGED_VARS"
  if [[ "$profile" == "default" ]]; then printf 'export CCP_PROFILE=%q\n' "default"; return 0; fi
  ccp_profile_exists "$home" "$profile" || { printf 'echo "⚠️  ccp: perfil %q no existe; default" >&2\n' "$profile"; printf 'export CCP_PROFILE=%q\n' "default"; return 0; }
  local type; type="$(ccp_profile_type "$home" "$profile")"
  case "$type" in
    official) printf 'export CLAUDE_CONFIG_DIR=%q\n' "$home/profiles/$profile/cc-home"
              printf 'export CCP_PROFILE=%q\n' "$profile" ;;
    deepseek) local b p f e k
              b="$(ccp_profile_get "$home" "$profile" base_url)"; p="$(ccp_profile_get "$home" "$profile" model_pro)"
              f="$(ccp_profile_get "$home" "$profile" model_flash)"; e="$(ccp_profile_get "$home" "$profile" effort)"
              printf 'export ANTHROPIC_BASE_URL=%q\n' "$b"
              if k="$(ccp_profile_get_key "$home" "$profile")"; then printf 'export ANTHROPIC_AUTH_TOKEN=%q\n' "$k"
              else printf 'echo "⚠️  ccp: %q sin key (ccp key %q)" >&2\n' "$profile" "$profile"; fi
              printf 'export ANTHROPIC_MODEL=%q\n' "$p"; printf 'export ANTHROPIC_DEFAULT_OPUS_MODEL=%q\n' "$p"
              printf 'export ANTHROPIC_DEFAULT_SONNET_MODEL=%q\n' "$p"; printf 'export ANTHROPIC_DEFAULT_HAIKU_MODEL=%q\n' "$f"
              printf 'export CLAUDE_CODE_SUBAGENT_MODEL=%q\n' "$f"; printf 'export CLAUDE_CODE_EFFORT_LEVEL=%q\n' "$e"
              printf 'export CCP_PROFILE=%q\n' "$profile" ;;
    *) printf 'export CCP_PROFILE=%q\n' "default" ;;
  esac; }

Heredoc _print_shell_init con la función ccp() (intercepta use/default/off/on/run, resto → command ccp) y el hook _ccp_autocheck (cache por PWD, eval "$(command ccp _hook "$PWD")"). cmd_install detecta y ofrece quitar el init viejo de dsctl; markers # >>> ccp shell init >>>. completion-shellinit imprime el heredoc (single source para tests + install).

ccp(){ case "$1" in
    use) shift; eval "$(command ccp _env "${1:-default}")" ;;
    default|off) eval "$(command ccp _env default)" ;;
    on)  eval "$(command ccp _env deepseek)" ;;   # legacy
    run) shift; eval "$(command ccp _hook "$PWD")"; if [[ $# -gt 0 ]]; then command "$@"; else command claude; fi ;;
    *)   command ccp "$@" ;; esac; }
_ccp_autocheck(){ command -v ccp >/dev/null 2>&1 || return
  [[ "$PWD" == "${_CCP_LAST_PWD:-}" ]] && return; _CCP_LAST_PWD="$PWD"
  eval "$(command ccp _hook "$PWD" 2>/dev/null)"; }
